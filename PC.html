Khởi tạo Database cho từng Microservice

1. Device Service MySQL 
-- Đây là cấu trúc chuẩn bạn đã thiết kế. 
-- Tôi rút gọn lại để làm ví dụ đại diện cho RDBMS quản lý thiết bị.

CREATE DATABASE IF NOT EXISTS device_service_db;
USE device_service_db;

CREATE TABLE stations (
    id VARCHAR(36) PRIMARY KEY,
    station_code VARCHAR(100) NOT NULL UNIQUE,
    name VARCHAR(255) NOT NULL,
    status VARCHAR(50) DEFAULT 'ACTIVE',
    created_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE sensors (
    id VARCHAR(36) PRIMARY KEY,
    station_id VARCHAR(36) NOT NULL,
    sensor_code VARCHAR(100) NOT NULL UNIQUE,
    min_value DOUBLE,
    max_value DOUBLE,
    CONSTRAINT fk_sensor_station FOREIGN KEY (station_id) REFERENCES stations(id)
);
2. Data Ingestion Service TimescaleDB (PostgreSQL) 
-- Sử dụng TimescaleDB (Mở rộng của PostgreSQL) chuyên dụng cho chuỗi thời gian.
-- Dữ liệu đẩy vào liên tục không cần khóa ngoại để tối ưu tốc độ Ghi (Write).

CREATE DATABASE ingestion_db;
\c ingestion_db;

-- Kích hoạt extension TimescaleDB
CREATE EXTENSION IF NOT EXISTS timescaledb;

CREATE TABLE sensor_data (
    time        TIMESTAMPTZ NOT NULL,
    sensor_id   VARCHAR(36) NOT NULL, -- ID từ Device Service
    value       DOUBLE PRECISION NOT NULL,
    battery     DOUBLE PRECISION
);

-- Biến bảng thông thường thành Hypertable để phân mảnh dữ liệu theo thời gian thực
SELECT create_hypertable('sensor_data', 'time');

-- Tạo Index để truy vấn theo cảm biến nhanh hơn
CREATE INDEX ix_sensor_time ON sensor_data (sensor_id, time DESC);
3. Alert / Rule Engine Service Redis + PostgreSQL 
/* ------ PHẦN 1: REDIS CACHE (Tra cứu siêu tốc) ------ */
// Lưu ngưỡng cảnh báo của sensor vào Redis dưới dạng Hash
HSET sensor_rule:SENS-001 min 10.5 max 50.0 is_active "true"

// Khi có data mới bắn tới, Service sẽ check cực nhanh:
HGETALL sensor_rule:SENS-001


/* ------ PHẦN 2: POSTGRESQL (Lưu lịch sử cảnh báo) ------ */
CREATE TABLE alert_history (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    sensor_id VARCHAR(36) NOT NULL,
    alert_type VARCHAR(50), -- 'OVER_TEMP', 'LOW_BATTERY', 'OFFLINE'
    recorded_value DOUBLE PRECISION,
    threshold_value DOUBLE PRECISION,
    message TEXT,
    severity VARCHAR(20), -- 'WARNING', 'CRITICAL'
    created_at TIMESTAMP DEFAULT NOW(),
    status VARCHAR(20) DEFAULT 'UNRESOLVED'
);

5. Identity & Auth Service MySQL 
-- Quản lý tài khoản và phân quyền (RBAC) chặt chẽ bằng quan hệ

CREATE TABLE users (
    id VARCHAR(36) PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    is_active BOOLEAN DEFAULT TRUE
);

CREATE TABLE roles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    role_name VARCHAR(50) NOT NULL UNIQUE -- 'ADMIN', 'VIEWER'
);

-- Bảng trung gian (N-N) giữa User và Role
CREATE TABLE user_roles (
    user_id VARCHAR(36),
    role_id INT,
    PRIMARY KEY (user_id, role_id),
    CONSTRAINT fk_ur_user FOREIGN KEY (user_id) REFERENCES users(id),
    CONSTRAINT fk_ur_role FOREIGN KEY (role_id) REFERENCES roles(id)
);
